// TODO: Implement parse
func parse(f *os.File) ([][]rune, error) {
    var m [][]rune
    scanner := bufio.NewScanner(f)
    for scanner.Scan() {
        line := scanner.Text()
        if line == "" {
            continue
        }
        m = append(m, []rune(line))
    }
    return m, nil
}
